		# Sample Logstash configuration for creating a simple
		# Beats -> Logstash -> Elasticsearch pipeline.

		input {
		  beats {
		    port => 5044
		  }
		  file {
			type => "java"
			path => "C:/log/dh-elk.log"
			codec => multiline {
				pattern => "^%{YEAR}-%{MONTHNUM}-%{MONTHDAY} % {TIME}.*"
				negate => "true"
				what => "previous"
			}
		  }
		}

		filter {
			if [message] =~ "\tat" {
				grok {
					match => ["message", "^(\tat)"]
					add_tag => ["stacktrace"]
				}
			}
		}

		output {
		  elasticsearch {
		    hosts => ["http://localhost:9200"]    
		  }
		  stdout {
			codec => rubydebug
		  }
		}
